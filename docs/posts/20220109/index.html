Hello World!

Hello World

     * プログラム作成、実行、確認のプロセスを知る
    */
    void Start()
    {

    }
複数行で実行
    /*
     * プログラムは上から順番に実行される
     * 表示する中身の数字は関係ない
    */
    void Start()
    {
        Debug.Log("3.Hello World!");
        Debug.Log("2.Hello World!");
        Debug.Log("1.Hello World!");
    }
実行エラー
    /*
     * エラー時の挙動を見てみる

    {
        // 1. 定義にない関数コール

        // 2. ; がない

変数について

変数の宣言
    /*
     * 変数は情報を保存するハコ
     * 宣言 -> 初期化 -> 利用の3段活用
     * (未)宣言と初期化は同時にできる
    */
    void Start()
    {
      // 宣言
      int number;
      // 代入(初期化)
      number = 10;
      // 利用ゥ!
      Debug.Log(number);
    }
計算した結果を変数に入れる
    /*
     * int型で計算ができる
     */
    void Start()
    {
      int number;
      // 算数の計算ができます！
      number = 10 + 10;
      Debug.Log(number);
    }
変数同士で代入
    /*
     * int型同士で代入ができる
     * 複数の変数を定義した時の挙動に注意
     */
    void Start()
    {
      int number;
      number = 10;
      int answer;
      // answerにnumberの数字を代入する
      answer = number;
      number = 20;
      // numberを20にしてもanswerは10のまま
      Debug.Log(answer);
    }
変数の宣言（エラー）
    /*
     * エラー時の挙動を見てみる
     * 1. 同じ名前で変数を２つ以上作れない
     * 2. (未)初期化を忘れない
     * 3. 宣言した「型」の物しか入れられない
    */
    // 1. 同じ変数を定義する
    void Start()
    {
      int number;
      int number;

      // 2. 宣言だけで初期化しない場合
      int number;
      Debug.Log(number);

      // 3. int型に文字を入れる
      int number = "abc";
    }
